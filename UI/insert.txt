    public void placeSigninPanel(JPanel panel){
        panel.setLayout(null);   
        
        JLabel nameLabel = new JLabel("Name");
        nameLabel.setBounds(10, 10, 80, 25);
        panel.add(nameLabel);
        
        JLabel idLabel = new JLabel("ID");
        idLabel.setBounds(10, 40, 80, 25);
        panel.add(idLabel);
        
        JLabel passLabel = new JLabel("Pass");
        passLabel.setBounds(10, 80, 80, 25);
        panel.add(passLabel);
        
        nameText = new JTextField(20);
        nameText.setBounds(100, 10, 160, 25);
        panel.add(nameText);
        
        idText = new JTextField(20);
        idText.setBounds(100, 40, 160, 25);
        panel.add(idText);
        
        btnPass = new JButton("입력");
        btnPass.setBounds(100, 80, 100, 25);
        panel.add(btnPass);
        btnPass.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
               makepass();
            }
        });
        
        btnOk = new JButton("저장");
        btnOk.setBounds(10, 130, 100, 25);
        panel.add(btnOk);
        btnOk.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent T) {
                
                // db에 있는 정보인지 확인/////////////////////
                try{
				BufferedWriter bos = new BufferedWriter(new FileWriter("회원명단.txt",true));
				bos.write(nameText.getText()+"/");
				bos.write(idText.getText()+"\r\n");
				bos.close();
				JOptionPane.showMessageDialog(null, "회원가입이 완료되었습니다.");
				dispose();
			}catch (Exception ex){
				JOptionPane.showMessageDialog(null, "회원가입에 실패하였습니다.");
			}
            }
        });
        
        btnCancel = new JButton("취소");
        btnCancel.setBounds(150, 130, 100, 25);
        panel.add(btnCancel);
        btnCancel.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
               dispose();
            }
        });
    }
